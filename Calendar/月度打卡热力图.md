```dataviewjs
// 配置
const diaryFolder = "Calendar";
const taskList = [
  "早上空腹服药",
  "六味地黄丸",
  "午餐前打针",
  "晚餐前打针",
  "睡前打针",
  "散步5000步"
];

// 自动获取当前年月
const now = new Date();
const currentYear = now.getFullYear();
const currentMonth = now.getMonth(); // 0~11，系统标准
const daysInMonth = new Date(currentYear, currentMonth + 1, 0).getDate();

// 加载日记
const pages = dv.pages(`"${diaryFolder}"`)
  .where(p => p.file.day)
  .sort(p => p.file.day, "asc");

// 构建日期 => 打卡完成情况
const dayStatus = {};
for (let day = 1; day <= daysInMonth; day++) {
  const dateStr = `${currentYear}-${String(currentMonth + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
  const page = pages.find(p => p.file.day.toFormat("yyyy-MM-dd") === dateStr);
  if (!page) {
    dayStatus[day] = { done: 0, total: 0 };
    continue;
  }
  const total = page.file.tasks.length;
  const done = page.file.tasks.filter(t => t.completed).length;
  dayStatus[day] = { done, total, ratio: total ? done / total : 0 };
}

// 绘制 GitHub 风格格子
dv.container.style.display = "grid";
dv.container.style.gap = "4px";
dv.container.style.gridTemplateColumns = "repeat(7, minmax(32px, 1fr))";
dv.container.style.marginTop = "1em";

// 星期表头
const weekLabels = ["一", "二", "三", "四", "五", "六", "日"];
weekLabels.forEach(w => {
  dv.el("div", w, { style: "text-align:center; font-weight:600; padding:4px;" });
});

// 1号是星期几（0=日，1=一，…6=六），转成周一为0
const firstDayOfMonth = (new Date(currentYear, currentMonth, 1).getDay() || 7) - 1;
// 前面补空位
for (let i = 0; i < firstDayOfMonth; i++) {
  dv.el("div", "", { style: "visibility:hidden;" });
}

// 日期格子
for (let day = 1; day <= daysInMonth; day++) {
  const { ratio } = dayStatus[day];
  let bg = "#eee";
  if (ratio >= 0.01) bg = "#d5f5d5";
  if (ratio >= 0.34) bg = "#9be9a8";
  if (ratio >= 0.67) bg = "#40c463";
  if (ratio >= 1.00) bg = "#216e39";

  const el = dv.el("div", day, {
    style: `
      width:32px; height:32px;
      background:${bg}; border-radius:4px;
      display:flex; align-items:center; justify-content:center;
      font-size:12px; cursor:pointer;
    `
  });
  el.onclick = () => {
    const fname = `${currentYear}-${String(currentMonth+1).padStart(2,0)}-${String(day).padStart(2,0)}.md`;
    app.workspace.openLinkText(fname, "", false);
  };
}

// 项目统计
dv.el("h4", `📊 ${currentYear}年${currentMonth+1}月打卡统计`);
taskList.forEach(item => {
  const count = pages.filter(p =>
    p.file.tasks.some(t => t.text.trim() === item && t.completed)
  ).length;
  dv.el("div", `- ${item}：完成 ${count} 天`);
});
```
