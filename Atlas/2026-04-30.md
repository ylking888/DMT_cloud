```dataviewjs
const folder = "Calendar";
const pages = dv.pages(`"${folder}"`)
  .where(p => p.file.day)
  .sort(p => p.file.day, 'asc');

const year = 2026;
const month = 4;
const daysInMonth = new Date(year, month + 1, 0).getDate();

// 打卡项目
const items = [
  "早上空腹服药",
  "六味地黄丸",
  "午餐前打针",
  "晚餐前打针",
  "睡前打针",
  "散步5000步"
];

// 生成日期→完成map
const dayMap = {};
for (let d = 1; d <= daysInMonth; d++) {
  const date = `${year}-${String(month).padStart(2, '0')}-${String(d).padStart(2, '0')}`;
  const page = pages.find(p => p.file.day.toFormat("yyyy-MM-dd") === date);
  dayMap[d] = {};
  if (page) {
    for (const item of items) {
      const task = page.file.tasks.find(t => t.text.trim() === item);
      dayMap[d][item] = task?.completed ? true : false;
    }
  }
}

// 绘制热力图
dv.container.style.display = "grid";
dv.container.style.gap = "4px";
dv.container.style.gridTemplateColumns = `repeat(7, minmax(30px, 1fr))`;
dv.container.style.marginTop = "1em";

// 表头星期
const week = ["一", "二", "三", "四", "五", "六", "日"];
week.forEach(day => {
  dv.el("div", day, { style: "text-align:center; font-weight:bold; padding:4px;" });
});

// 填充日期格子
for (let d = 1; d <= daysInMonth; d++) {
  const date = `${year}-${String(month).padStart(2, '0')}-${String(d).padStart(2, '0')}`;
  const page = pages.find(p => p.file.day.toFormat("yyyy-MM-dd") === date);
  const done = page ? page.file.tasks.filter(t => t.completed).length : 0;
  const total = page ? page.file.tasks.length : 0;
  const ratio = total > 0 ? done / total : 0;

  let color = "#eee";
  if (ratio > 0) color = "#d5f5d5";
  if (ratio > 0.3) color = "#9be9a8";
  if (ratio > 0.6) color = "#40c463";
  if (ratio >= 1) color = "#216e39";

  const div = dv.el("div", d, {
    style: `
      width:30px; height:30px;
      background:${color};
      border-radius:4px;
      display:flex;align-items:center;justify-content:center;
      font-size:12px;cursor:pointer;
    `
  });
  div.onclick = () => app.workspace.openLinkText(`${date}.md`, "", false);
}

// 项目明细统计
dv.header(3, "项目完成明细");
for (const item of items) {
  const count = pages.filter(p =>
    p.file.tasks.some(t => t.text.trim() === item && t.completed)
  ).length;
  dv.el("div", `- ${item}：本月完成 **${count}** 天`);
}
```
